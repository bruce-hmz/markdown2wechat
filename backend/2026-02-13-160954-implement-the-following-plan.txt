│ from pygments import highlight                                               │

│ from pygments.formatters import HtmlFormatter                                │
│                                                                              │
│ def highlight_code(code: str, language: str) -> str:                         │
│     """高亮代码并返回内联样式的 HTML"""                                      │
│     lexer = get_lexer_by_name(language, stripall=True)                       │
│     formatter = HtmlFormatter(noclasses=True, nowrap=True)                   │
│     return highlight(code, lexer, formatter)                                 │

│ 3. HTML 清理（移除不支持的标签）                                             │
│ from bs4 import BeautifulSoup                                                │
│                                                                              │
│ SUPPORTED_TAGS = {                                                           │
│     'p', 'br', 'div', 'span', 'h1', 'h2', 'h3', 'h4', 'h5', 'h6',            │
│     'strong', 'em', 'b', 'i', 'u', 's', 'blockquote',                        │
│     'ul', 'ol', 'li', 'img', 'a', 'section'                                  │
│ }                                                                            │
│                                                                              │
│ def sanitize_html(html: str) -> str:                                         │
│     """清理不支持的 HTML 标签"""                                             │
│     soup = BeautifulSoup(html, 'lxml')                                       │
│     for tag in soup.find_all(True):                                          │
│         if tag.name not in SUPPORTED_TAGS:                                   │
│             tag.unwrap()                                                     │
│     return str(soup)                                                         │

│ 4. 实现图片处理服务                                                          │
│                                                                              │
│ 目标： 自动处理图片（压缩、Base64 转换、尺寸优化）                           │
│                                                                              │
│ 关键文件：                                                                   │
│ - backend/app/services/image_service.py:1-120 - 图片处理核心                 │
│                                                                              │
│ 处理流程：                                                                   │
│ 1. 检查图片大小                                                              │
│ 2. 压缩质量优化（默认 85%）                                                  │
│ 3. 调整尺寸（最大宽度 750px，适配移动端）                                    │
│ 4. 转换为 Base64（小图片 < 50KB）或返回 URL                                  │
│                                                                              │
│ 示例代码：                                                                   │
│ from PIL import Image                                                        │
│ import io, base64                                                            │
│                                                                              │
│ async def process_image(image_file, max_width=750) -> dict:                  │
│     """处理并优化图片"""                                                     │
│     img = Image.open(image_file)                                             │
│                                                                              │
│     # 调整尺寸                                                               │
│     if img.width > max_width:                                                │
│         ratio = max_width / img.width                                        │
│         img = img.resize((max_width, int(img.height * ratio)))               │
│                                                                              │
│     # 转换为 Base64                                                          │
│     buffer = io.BytesIO()                                                    │
│     img.save(buffer, format='JPEG', quality=85)                              │
│     base64_str = base64.b64encode(buffer.getvalue()).decode()                │
│                                                                              │
│     return {                                                                 │
│         "base64": f"data:image/jpeg;base64,{base64_str}",                    │
│         "size": len(buffer.getvalue())                                       │
│     }                                                                        │

│ WebSocket 实时转换示例：                                                     │
│ @router.websocket("/ws/convert")                                             │
│ async def websocket_convert(websocket: WebSocket):                           │
│     """实时转换（边写边看）"""                                               │
│     await websocket.accept()                                                 │
│                                                                              │
│     try:                                                                     │
│         while True:                                                          │
│             data = await websocket.receive_json()                            │
│                                                                              │
│             # 实时转换                                                       │
│             service = MarkdownService()                                      │
│             theme = ThemeService().load_theme(data.get('theme', 'default'))  │
│             html = service.convert_to_wechat_html(data['markdown'], theme)   │
│                                                                              │
│             await websocket.send_json({"html": html})                        │
│     except WebSocketDisconnect:                                              │
│         pass                                                                 │
